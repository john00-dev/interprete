// ====== USO SEGURO CON OBJETOS ======

funcion mostrarUsuario() {

    si (usuario != indefinido) {
        consola.escribir(usuario.nombre);
    } sino {
        consola.escribir("Usuario aún no definido");
    }
}


mostrarUsuario();

var usuario = {
    nombre: "Carlos"
};


mostrarUsuario();
